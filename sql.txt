
CREATE TABLE `ups` (
`id` INT UNSIGNED NOT NULL AUTO_INCREMENT ,
`name` VARCHAR( 250 ) NOT NULL ,
`vip` VARCHAR( 15 ) NOT NULL ,
`dl` BIGINT UNSIGNED NOT NULL ,
`remup` TINYINT NOT NULL DEFAULT '0',
`durl` VARCHAR( 2048 ) NOT NULL ,
`size` INT UNSIGNED NOT NULL ,
PRIMARY KEY ( `id` ) ) CHARACTER SET utf8 COLLATE utf8_persian_ci;


CREATE TABLE `timer` (
`id` INT UNSIGNED NOT NULL AUTO_INCREMENT PRIMARY KEY ,
`vip` VARCHAR( 15 ) CHARACTER SET utf8 COLLATE utf8_persian_ci NOT NULL ,
`timer` BIGINT UNSIGNED NOT NULL
) CHARACTER SET utf8 COLLATE utf8_persian_ci;
